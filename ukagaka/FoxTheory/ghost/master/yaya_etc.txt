OnInstallBegin
{
  <<"
  リサ６７：あれ、荷物？
  リサ０５：誰からだろう……
  ">>
}

//------------------------------------------------------------------------------
//インストール完了メッセージ
//------------------------------------------------------------------------------
OnInstallComplete
{
	reference[0] = SPLIT(reference[0],' with',2)[0] //hogehoge with hemohemo
	ref_id = ''
	if reference[0] == "shell" {
		ref_id = 'shell'
	}
	elseif reference[0] == "ghost" {
		ref_id = 'ghost'
	}
	elseif reference[0] == "balloon" {
		ref_id = 'balloon'
	}

	<<"
	リサ０１【R0 == 'shell'】：シェル、
	リサ０１【R0 == 'ghost'】：ゴースト、
	リサ０１【R0 == 'balloon'】：バルーン、
	リサ０１【R0 == 'plugin'】：プラグイン、
	リサ０６【それ以外】：Undefined、
	改行無しリサ：「%(reference[1])」、インストール完了。
	リサ【STRLEN(ref_id) > 0】：切り替える？
	＠メニュー【直前】：切り替える｜OnAYATEMPLATE.InstallChange｜%(ref_id)
	＠メニュー【直前】：切り替えない｜Menu_CANCEL
	">>
}

OnAYATEMPLATE.InstallChange
{
	<<"
	リサ０５：Please wait a moment...\![change,%(reference[0]),lastinstalled]
	">>
}
